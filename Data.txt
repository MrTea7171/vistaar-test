Create a mongo query to list down account ids which has made at least one transaction below the amount 5000

  {
    $group: {
      _id: "$account_id",
      transactionsBelow5000: {
        $sum: {
          $cond: [{ $lt: ["$amount", 5000] }, 1, 0]
        }
      }
    }
  },

    }
  },
  {
    $project: {

    }
  }
